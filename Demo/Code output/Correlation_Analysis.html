## Plot correlation diagram
import matplotlib.pyplot as plt
%matplotlib inline
df2 = df
correlations = df2.corr()
fig = plt.figure(figsize=(10, 8))
ax = fig.add_subplot(111)
cax = ax.matshow(correlations, vmin=-1, vmax=1)
fig.colorbar(cax)
ticks = numpy.arange(0,12,1)
ax.set_xticks(ticks)
ax.set_yticks(ticks)
ax.set_xticklabels(df2.columns, fontsize=6)
ax.set_yticklabels(df2.columns)
plt.show()
